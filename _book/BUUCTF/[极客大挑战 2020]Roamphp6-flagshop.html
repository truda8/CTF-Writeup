自动计算：

# -*- coding: UTF-8 -*-
import hashlib

def getMd5(index):
    for i in range(100000, 100000000):
        x = i
        md5 = hashlib.md5(str(x).encode("utf8")).hexdigest()
        if md5[0:5] == index:
            return x
print(getMd5("a368a"))
